## üêç **–°–æ–∑–¥–∞–µ–º scripts/data-processing.py:**

```python
#!/usr/bin/env python3
"""
–°–∫—Ä–∏–ø—Ç –æ–±—Ä–∞–±–æ—Ç–∫–∏ –¥–∞–Ω–Ω—ã—Ö –¥–ª—è Asia Logistics
–û–±—Ä–∞–±–æ—Ç–∫–∞, –æ—á–∏—Å—Ç–∫–∞ –∏ –ø–æ–¥–≥–æ—Ç–æ–≤–∫–∞ –¥–∞–Ω–Ω—ã—Ö –¥–ª—è –∞–Ω–∞–ª–∏–∑–∞
"""

import pandas as pd
import numpy as np
from datetime import datetime, timedelta
import logging
from typing import Dict, List, Optional

# –ù–∞—Å—Ç—Ä–æ–π–∫–∞ –ª–æ–≥–∏—Ä–æ–≤–∞–Ω–∏—è
logging.basicConfig(level=logging.INFO)
logger = logging.getLogger(__name__)

class DataProcessor:
    """–ö–ª–∞—Å—Å –¥–ª—è –æ–±—Ä–∞–±–æ—Ç–∫–∏ –¥–∞–Ω–Ω—ã—Ö –ª–æ–≥–∏—Å—Ç–∏–∫–∏"""
    
    def __init__(self):
        self.processed_data = {}
    
    def load_shipment_data(self, file_path: str) -> pd.DataFrame:
        """–ó–∞–≥—Ä—É–∑–∫–∞ –¥–∞–Ω–Ω—ã—Ö –æ –ø–æ—Å—Ç–∞–≤–∫–∞—Ö"""
        try:
            df = pd.read_csv(file_path, parse_dates=['created_at', 'actual_delivery'])
            logger.info(f"–ó–∞–≥—Ä—É–∂–µ–Ω–æ {len(df)} –∑–∞–ø–∏—Å–µ–π –æ –ø–æ—Å—Ç–∞–≤–∫–∞—Ö")
            return df
        except Exception as e:
            logger.error(f"–û—à–∏–±–∫–∞ –∑–∞–≥—Ä—É–∑–∫–∏ –¥–∞–Ω–Ω—ã—Ö: {e}")
            raise
    
    def clean_shipment_data(self, df: pd.DataFrame) -> pd.DataFrame:
        """–û—á–∏—Å—Ç–∫–∞ –∏ –ø—Ä–µ–¥–æ–±—Ä–∞–±–æ—Ç–∫–∞ –¥–∞–Ω–Ω—ã—Ö"""
        # –£–¥–∞–ª–µ–Ω–∏–µ –¥—É–±–ª–∏–∫–∞—Ç–æ–≤
        df = df.drop_duplicates(subset=['tracking_number'])
        
        # –ó–∞–ø–æ–ª–Ω–µ–Ω–∏–µ –ø—Ä–æ–ø—É—â–µ–Ω–Ω—ã—Ö –∑–Ω–∞—á–µ–Ω–∏–π
        df['weight_kg'] = df['weight_kg'].fillna(df['weight_kg'].median())
        df['volume_m3'] = df['volume_m3'].fillna(df['volume_m3'].median())
        
        # –†–∞—Å—á–µ—Ç –≤—Ä–µ–º–µ–Ω–∏ –¥–æ—Å—Ç–∞–≤–∫–∏
        df['delivery_days'] = (
            df['actual_delivery'] - df['created_at']
        ).dt.total_seconds() / (24 * 3600)
        
        # –§–∏–ª—å—Ç—Ä–∞—Ü–∏—è –∞–Ω–æ–º–∞–ª—å–Ω—ã—Ö –∑–Ω–∞—á–µ–Ω–∏–π
        df = df[(df['delivery_days'] > 0) & (df['delivery_days'] < 30)]
        
        logger.info("–î–∞–Ω–Ω—ã–µ —É—Å–ø–µ—à–Ω–æ –æ—á–∏—â–µ–Ω—ã")
        return df
    
    def calculate_kpis(self, df: pd.DataFrame) -> Dict:
        """–†–∞—Å—á–µ—Ç –∫–ª—é—á–µ–≤—ã—Ö –ø–æ–∫–∞–∑–∞—Ç–µ–ª–µ–π —ç—Ñ—Ñ–µ–∫—Ç–∏–≤–Ω–æ—Å—Ç–∏"""
        kpis = {
            'total_shipments': len(df),
            'on_time_deliveries': len(df[df['delivery_days'] <= 5]),
            'average_delivery_time': df['delivery_days'].mean(),
            'median_delivery_time': df['delivery_days'].median(),
            'delivery_time_std': df['delivery_days'].std(),
            'on_time_rate': len(df[df['delivery_days'] <= 5]) / len(df)
        }
        
        # –ê–Ω–∞–ª–∏–∑ –ø–æ –∫–∞—Ç–µ–≥–æ—Ä–∏—è–º
        if 'product_category' in df.columns:
            category_kpis = df.groupby('product_category').agg({
                'delivery_days': ['mean', 'std', 'count'],
                'weight_kg': 'mean'
            }).round(2)
            kpis['by_category'] = category_kpis.to_dict()
        
        logger.info("KPIs —É—Å–ø–µ—à–Ω–æ —Ä–∞—Å—Å—á–∏—Ç–∞–Ω—ã")
        return kpis
    
    def detect_anomalies(self, df: pd.DataFrame) -> pd.DataFrame:
        """–û–±–Ω–∞—Ä—É–∂–µ–Ω–∏–µ –∞–Ω–æ–º–∞–ª–∏–π –≤ –¥–∞–Ω–Ω—ã—Ö"""
        from sklearn.ensemble import IsolationForest
        
        # –ü—Ä–∏–∑–Ω–∞–∫–∏ –¥–ª—è –æ–±–Ω–∞—Ä—É–∂–µ–Ω–∏—è –∞–Ω–æ–º–∞–ª–∏–π
        features = ['delivery_days', 'weight_kg', 'volume_m3']
        X = df[features].fillna(0)
        
        # –û–±—É—á–µ–Ω–∏–µ –º–æ–¥–µ–ª–∏ –æ–±–Ω–∞—Ä—É–∂–µ–Ω–∏—è –∞–Ω–æ–º–∞–ª–∏–π
        iso_forest = IsolationForest(contamination=0.05, random_state=42)
        anomalies = iso_forest.fit_predict(X)
        
        df['is_anomaly'] = anomalies == -1
        df['anomaly_score'] = iso_forest.decision_function(X)
        
        logger.info(f"–û–±–Ω–∞—Ä—É–∂–µ–Ω–æ {df['is_anomaly'].sum()} –∞–Ω–æ–º–∞–ª–∏–π")
        return df
    
    def prepare_time_series(self, df: pd.DataFrame, freq: str = 'D') -> pd.DataFrame:
        """–ü–æ–¥–≥–æ—Ç–æ–≤–∫–∞ –¥–∞–Ω–Ω—ã—Ö –¥–ª—è –≤—Ä–µ–º–µ–Ω–Ω—ã—Ö —Ä—è–¥–æ–≤"""
        df_ts = df.set_index('created_at').resample(freq).agg({
            'tracking_number': 'count',
            'delivery_days': 'mean',
            'weight_kg': 'sum',
            'is_anomaly': 'sum'
        }).rename(columns={
            'tracking_number': 'daily_shipments',
            'delivery_days': 'avg_delivery_time',
            'weight_kg': 'total_weight',
            'is_anomaly': 'anomaly_count'
        })
        
        # –ó–∞–ø–æ–ª–Ω–µ–Ω–∏–µ –ø—Ä–æ–ø—É—Å–∫–æ–≤ –≤ –≤—Ä–µ–º–µ–Ω–Ω—ã—Ö —Ä—è–¥–∞—Ö
        df_ts = df_ts.fillna(method='ffill')
        
        return df_ts
    
    def generate_report(self, kpis: Dict, output_path: str):
        """–ì–µ–Ω–µ—Ä–∞—Ü–∏—è –æ—Ç—á–µ—Ç–∞ —Å –∞–Ω–∞–ª–∏–∑–æ–º"""
        report = f"""
# –û—Ç—á–µ—Ç –∞–Ω–∞–ª–∏–∑–∞ –¥–∞–Ω–Ω—ã—Ö Asia Logistics
## –ü–µ—Ä–∏–æ–¥: {datetime.now().strftime('%Y-%m-%d')}

### –ö–ª—é—á–µ–≤—ã–µ –ø–æ–∫–∞–∑–∞—Ç–µ–ª–∏:
- –í—Å–µ–≥–æ –ø–æ—Å—Ç–∞–≤–æ–∫: {kpis['total_shipments']}
- –ü–æ—Å—Ç–∞–≤–∫–∏ –≤ —Å—Ä–æ–∫: {kpis['on_time_deliveries']}
- –°—Ä–µ–¥–Ω–µ–µ –≤—Ä–µ–º—è –¥–æ—Å—Ç–∞–≤–∫–∏: {kpis['average_delivery_time']:.2f} –¥–Ω–µ–π
- –ü—Ä–æ—Ü–µ–Ω—Ç –ø–æ—Å—Ç–∞–≤–æ–∫ –≤ —Å—Ä–æ–∫: {kpis['on_time_rate']:.1%}

### –†–µ–∫–æ–º–µ–Ω–¥–∞—Ü–∏–∏:
"""
        
        # –î–æ–±–∞–≤–ª–µ–Ω–∏–µ —Ä–µ–∫–æ–º–µ–Ω–¥–∞—Ü–∏–π –Ω–∞ –æ—Å–Ω–æ–≤–µ –∞–Ω–∞–ª–∏–∑–∞
        if kpis['on_time_rate'] < 0.85:
            report += "- –£–≤–µ–ª–∏—á–∏—Ç—å –º–æ–Ω–∏—Ç–æ—Ä–∏–Ω–≥ –∑–∞–¥–µ—Ä–∂–µ–∫ –ø–æ—Å—Ç–∞–≤–æ–∫\n"
        
        if kpis['delivery_time_std'] > 2.0:
            report += "- –°—Ç–∞–Ω–¥–∞—Ä—Ç–∏–∑–∏—Ä–æ–≤–∞—Ç—å –ø—Ä–æ—Ü–µ—Å—Å—ã –¥–æ—Å—Ç–∞–≤–∫–∏\n"
        
        # –°–æ—Ö—Ä–∞–Ω–µ–Ω–∏–µ –æ—Ç—á–µ—Ç–∞
        with open(output_path, 'w', encoding='utf-8') as f:
            f.write(report)
        
        logger.info(f"–û—Ç—á–µ—Ç —Å–æ—Ö—Ä–∞–Ω–µ–Ω: {output_path}")

def main():
    """–û—Å–Ω–æ–≤–Ω–∞—è —Ñ—É–Ω–∫—Ü–∏—è –≤—ã–ø–æ–ª–Ω–µ–Ω–∏—è —Å–∫—Ä–∏–ø—Ç–∞"""
    processor = DataProcessor()
    
    try:
        # –ó–∞–≥—Ä—É–∑–∫–∞ –¥–∞–Ω–Ω—ã—Ö
        df = processor.load_shipment_data('data/shipments.csv')
        
        # –û—á–∏—Å—Ç–∫–∞ –¥–∞–Ω–Ω—ã—Ö
        df_clean = processor.clean_shipment_data(df)
        
        # –†–∞—Å—á–µ—Ç KPI
        kpis = processor.calculate_kpis(df_clean)
        
        # –û–±–Ω–∞—Ä—É–∂–µ–Ω–∏–µ –∞–Ω–æ–º–∞–ª–∏–π
        df_anomalies = processor.detect_anomalies(df_clean)
        
        # –ü–æ–¥–≥–æ—Ç–æ–≤–∫–∞ –≤—Ä–µ–º–µ–Ω–Ω—ã—Ö —Ä—è–¥–æ–≤
        df_ts = processor.prepare_time_series(df_anomalies)
        
        # –ì–µ–Ω–µ—Ä–∞—Ü–∏—è –æ—Ç—á–µ—Ç–∞
        processor.generate_report(kpis, 'reports/data_analysis_report.md')
        
        # –°–æ—Ö—Ä–∞–Ω–µ–Ω–∏–µ –æ–±—Ä–∞–±–æ—Ç–∞–Ω–Ω—ã—Ö –¥–∞–Ω–Ω—ã—Ö
        df_clean.to_csv('processed_data/cleaned_shipments.csv', index=False)
        df_ts.to_csv('processed_data/time_series_data.csv')
        
        logger.info("–û–±—Ä–∞–±–æ—Ç–∫–∞ –¥–∞–Ω–Ω—ã—Ö –∑–∞–≤–µ—Ä—à–µ–Ω–∞ —É—Å–ø–µ—à–Ω–æ")
        
    except Exception as e:
        logger.error(f"–û—à–∏–±–∫–∞ –≤ –æ—Å–Ω–æ–≤–Ω–æ–º –ø—Ä–æ—Ü–µ—Å—Å–µ: {e}")

if __name__ == "__main__":
    main()
